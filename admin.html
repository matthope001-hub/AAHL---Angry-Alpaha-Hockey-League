Shared UtilitiesAll agents use these
const SS = SpreadsheetApp.getActiveSpreadsheet();
const CONFIG = () => {
  const cfg = SS.getSheetByName('Config').getDataRange().getValues();
  return Object.fromEntries(cfg.map(r => [r[0], r[1]]));
};
function agentLog(agentId, status, message, requiresAction = false) {
  SS.getSheetByName('AgentStatus').appendRow([new Date(), agentId, status, message, requiresAction ? 'Y' : 'N']);
}
function agentReady(agentId) {
  const rows = SS.getSheetByName('AgentStatus').getDataRange().getValues();
  const today = Utilities.formatDate(new Date(), 'EST', 'yyyy-MM-dd');
  const todayRows = rows.filter(r => Utilities.formatDate(new Date(r[0]), 'EST', 'yyyy-MM-dd') === today && r[1] === agentId);
  if (!todayRows.length) return false;
  return todayRows[todayRows.length - 1][2] === '✅';
}
function alertCommissioner(subject, body) { MailApp.sendEmail(CONFIG()['CommissionerEmail'], subject, body); }
function gameAlreadyProcessed(gameId) { return SS.getSheetByName('ProcessedGames').getDataRange().getValues().some(r => String(r[0]) === String(gameId)); }
function markGameProcessed(gameId, date, home, away) { SS.getSheetByName('ProcessedGames').appendRow([gameId, date, home, away, new Date()]); }
AGT-004 — Stat Fetch AgentNightly 3 AM
function fetchNightlyStats() {
  agentLog('AGT-004', '⏳', 'Starting nightly stat fetch');
  const date = Utilities.formatDate(new Date(Date.now() - 86400000), 'EST', 'yyyy-MM-dd');
  try {
    const sched = JSON.parse(UrlFetchApp.fetch(`https://api-web.nhle.com/v1/schedule/${date}`).getContentText());
    const games = sched.gameWeek?.[0]?.games || [];
    if (!games.length) { agentLog('AGT-004', '✅', `No games on ${date}`); return; }
    let processed = 0, skipped = 0, anomalies = 0;
    games.forEach(game => {
      if (gameAlreadyProcessed(game.id)) { skipped++; return; }
      const box = JSON.parse(UrlFetchApp.fetch(`https://api-web.nhle.com/v1/gamecenter/${game.id}/boxscore`).getContentText());
      const homeGoals = box.homeTeam?.score || 0, awayGoals = box.awayTeam?.score || 0;
      const isOT = ['OT','SO'].includes(box.periodDescriptor?.periodType);
      ['homeTeam','awayTeam'].forEach(side => {
        const team = box[side]; if (!team) return;
        [...(team.forwards||[]), ...(team.defense||[])].forEach(p => {
          if (!p.toi) return;
          const stats = { name: p.name?.default||'', team: team.abbrev, pos: team.forwards?.includes(p)?'F':'D', goals: p.goals||0, assists: p.assists||0, sog: p.sog||0, pim: p.pim||0 };
          if (stats.goals > 6 || stats.assists > 6) { anomalies++; agentLog('AGT-006','⚠️',`Suspicious: ${stats.name} — ${stats.goals}G ${stats.assists}A`,true); return; }
          updateSkaterStats(stats);
          checkHatTrick(stats, game.id, date);
        });
        (team.goalies||[]).forEach(g => {
          if (!g.toi || g.toi==='00:00') return;
          const teamWon = (side==='homeTeam') ? homeGoals>awayGoals : awayGoals>homeGoals;
          updateGoalieStats({ name: g.name?.default||'', team: team.abbrev, pos:'G', wins: teamWon&&!isOT?1:0, losses: !teamWon&&!isOT?1:0, otl: !teamWon&&isOT?1:0, shutouts: g.shotsAgainst>0&&g.goalsAgainst===0?1:0, saves: g.saves||0 });
        });
      });
      markGameProcessed(game.id, date, box.homeTeam?.abbrev, box.awayTeam?.abbrev);
      processed++;
    });
    agentLog('AGT-004','✅',`Complete — ${processed} processed, ${skipped} skipped, ${anomalies} anomalies`);
    if (anomalies > 0) alertCommissioner('AAHL ⚠️ Anomalies', `${anomalies} suspicious stat(s) flagged on ${date}`);
    calculateStandings();
  } catch(err) { agentLog('AGT-004','❌',`Fatal: ${err.message}`,true); alertCommissioner('AAHL ❌ Stat Fetch Failed', err.message); }
}
AGT-005 — Standings AgentRuns after AGT-004
function calculateStandings() {
  if (!agentReady('AGT-004')) { agentLog('AGT-005','⏸','Waiting — AGT-004 not complete'); return; }
  agentLog('AGT-005','⏳','Recalculating standings');
  try {
    const players = getPlayerMap(), entries = getEntries();
    const ranked = entries.map(e => {
      const pts = Math.round(e.picks.reduce((s,n)=>s+(players[n]?.pts||0),0)*100)/100;
      return { ...e, pts, goals: e.picks.reduce((s,n)=>s+(players[n]?.goals||0),0), assists: e.picks.reduce((s,n)=>s+(players[n]?.assists||0),0), saves: e.picks.reduce((s,n)=>s+(players[n]?.saves||0),0) };
    }).sort((a,b)=>b.pts-a.pts||(b.goals+b.assists)-(a.goals+a.assists)||b.saves-a.saves);
    writeStandings(ranked);
    agentLog('AGT-005','✅',`${ranked.length} entries ranked. Leader: ${ranked[0]?.teamName} (${ranked[0]?.pts} pts)`);
    if (new Date().getDay()===1) sendWeeklyDigest(ranked);
  } catch(err) { agentLog('AGT-005','❌',err.message,true); }
}
AGT-007 — Comms AgentMonday digest + on-demand
function sendEntryConfirmation(entry) {
  MailApp.sendEmail({ to: entry.email, subject: '🏒 AAHL Entry Confirmed — ' + entry.teamName,
    htmlBody: `<h2>You're in, ${entry.firstName}!</h2><p>Your entry <strong>${entry.teamName}</strong> has been received.</p><p>Send $10 to matt.hope@rocketmail.com. Picks lock September 29, 2026.</p>` });
  agentLog('AGT-007','✅',`Confirmation sent to ${entry.email}`);
}
function sendWeeklyDigest(standings) {
  if (!agentReady('AGT-005')) { agentLog('AGT-007','⏸','Holding — AGT-005 not ready'); return; }
  agentLog('AGT-007','⏳','Building weekly digest');
  const allEntries = getEntries();
  const rows = standings.map((e,i)=>`<tr><td>${i+1}</td><td>${e.teamName}</td><td>${e.pts}</td></tr>`).join('');
  allEntries.forEach(e => MailApp.sendEmail({ to: e.email, subject: `🦙 AAHL Weekly Update — ${standings[0].teamName} leads`,
    htmlBody: `<h2>AAHL Standings</h2><table border="1" cellpadding="6" style="border-collapse:collapse"><tr><th>#</th><th>Team</th><th>Pts</th></tr>${rows}</table>` }));
  agentLog('AGT-007','✅',`Digest sent to ${allEntries.length} participants`);
}
AGT-002 — Entry ReceiverOn form submit
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents), cfg = CONFIG();
    if (cfg['PicksLocked']===true || cfg['PicksLocked']==='TRUE')
      return ContentService.createTextOutput(JSON.stringify({success:false,error:'Picks are locked.'})).setMimeType(ContentService.MimeType.JSON);
    const sheet = SS.getSheetByName('Entries'), id = 'AAHL-'+Date.now();
    sheet.appendRow([id, data.firstName, data.lastName, data.email, data.teamName, 'N', new Date(), ...data.picks]);
    agentLog('AGT-002','✅',`New entry: ${data.teamName} by ${data.firstName} ${data.lastName}`);
    sendEntryConfirmation(data);
    return ContentService.createTextOutput(JSON.stringify({success:true,entryId:id})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    agentLog('AGT-002','❌',err.message,true);
    return ContentService.createTextOutput(JSON.stringify({success:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
