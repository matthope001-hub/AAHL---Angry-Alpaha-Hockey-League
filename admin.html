const SS = SpreadsheetApp.getActiveSpreadsheet();
const CONFIG = () => {
  const cfg = SS.getSheetByName('Config').getDataRange().getValues();
  return Object.fromEntries(cfg.map(r => [r[0], r[1]]));
};

// ── Agent Status Logger ──
function agentLog(agentId, status, message, requiresAction = false) {
  SS.getSheetByName('AgentStatus').appendRow([
    new Date(), agentId, status, message, requiresAction ? 'Y' : 'N'
  ]);
}

// ── Check if a previous agent completed successfully today ──
function agentReady(agentId) {
  const rows = SS.getSheetByName('AgentStatus').getDataRange().getValues();
  const today = Utilities.formatDate(new Date(), 'EST', 'yyyy-MM-dd');
  const todayRows = rows.filter(r =>
    Utilities.formatDate(new Date(r[0]), 'EST', 'yyyy-MM-dd') === today &&
    r[1] === agentId
  );
  if (!todayRows.length) return false;
  return todayRows[todayRows.length - 1][2] === '✅';
}

// ── Alert commissioner ──
function alertCommissioner(subject, body) {
  const cfg = CONFIG();
  MailApp.sendEmail(cfg['CommissionerEmail'], subject, body);
}

// ── Duplicate game guard ──
function gameAlreadyProcessed(gameId) {
  const rows = SS.getSheetByName('ProcessedGames').getDataRange().getValues();
  return rows.some(r => String(r[0]) === String(gameId));
}

function markGameProcessed(gameId, date, home, away, homeScore, awayScore) {
  SS.getSheetByName('ProcessedGames').appendRow([
    gameId, date, home, away, homeScore, awayScore, new Date()
  ]);
}

// ── Pool player name lookup ──
function getPoolPlayerNames() {
  const rows = SS.getSheetByName('Players').getDataRange().getValues();
  return new Set(rows.slice(1).map(r => (r[0] || '').trim()));
}

// ── Read all entries from sheet ──
function getEntries() {
  const rows = SS.getSheetByName('Entries').getDataRange().getValues();
  return rows.slice(1).map(r => ({
    entryId:         r[0],
    firstName:       r[1],
    lastName:        r[2],
    email:           r[3],
    phone:           r[4],
    teamName:        r[5],
    payment:         r[6],
    picks:           r.slice(8, 32),
    divAtlantic:     (r[32] || '').toString().trim(),
    divMetropolitan: (r[33] || '').toString().trim(),
    divCentral:      (r[34] || '').toString().trim(),
    divPacific:      (r[35] || '').toString().trim(),
  })).filter(e => e.entryId);
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const cfg = CONFIG();

    if (cfg['PicksLocked'] === true || cfg['PicksLocked'] === 'TRUE') {
      return ContentService.createTextOutput(
        JSON.stringify({ success: false, error: 'Picks are locked.' })
      ).setMimeType(ContentService.MimeType.JSON);
    }

    const sheet = SS.getSheetByName('Entries');
    const id = 'AAHL-' + Utilities.formatDate(new Date(), 'EST', 'yyyyMMdd') + '-' +
               Math.floor(Math.random() * 90000 + 10000);

    sheet.appendRow([
      id,
      data.firstName, data.lastName,
      data.email,
      data.phone     || '',
      data.teamName,
      'N',
      new Date(),
      ...data.picks,
      data.divisions?.atlantic     || '',
      data.divisions?.metropolitan || '',
      data.divisions?.central      || '',
      data.divisions?.pacific      || '',
    ]);

    agentLog('AGT-001', '✅', `New entry: ${data.teamName} — ${data.firstName} ${data.lastName} (${data.email})`);
    sendEntryConfirmation(data);

    return ContentService.createTextOutput(
      JSON.stringify({ success: true, entryId: id })
    ).setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    agentLog('AGT-001', '❌', `Entry submission failed: ${err.message}`, true);
    return ContentService.createTextOutput(
      JSON.stringify({ success: false, error: err.message })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

// AGT-002 — Stat Fetch + Anomaly Detection (Nightly 3 AM)
// ══════════════════════════════════════════════
function fetchNightlyStats() {
  agentLog('AGT-002', '⏳', 'Starting nightly stat fetch');
  const date = Utilities.formatDate(
    new Date(Date.now() - 86400000), 'EST', 'yyyy-MM-dd'
  );

  const poolPlayers = getPoolPlayerNames();

  try {
    const sched = JSON.parse(UrlFetchApp
      .fetch(`https://api-web.nhle.com/v1/schedule/${date}`)
      .getContentText());

    const games = sched.gameWeek?.[0]?.games || [];
    if (!games.length) {
      agentLog('AGT-002', '✅', `No games on ${date} — nothing to process`);
      calculateStandings();
      return;
    }

    let processed = 0, skipped = 0, anomalies = 0;

    const nightlySheet = SS.getSheetByName('NightlyStats');
    nightlySheet.clearContents();
    nightlySheet.appendRow([
      'PlayerName','Team','Pos','Game','Goals','Assists','SOG','PIM',
      'Wins','Losses','OTL','Shutouts','Saves','PtsEarned','Date'
    ]);

    const nightlyRows = [];
    const teamsPlayed = new Set();

    games.forEach(game => {
      if (gameAlreadyProcessed(game.id)) {
        skipped++;
        agentLog('AGT-002', '⚠️', `Game ${game.id} already processed — skipping`, false);
        return;
      }

      const box = JSON.parse(UrlFetchApp
        .fetch(`https://api-web.nhle.com/v1/gamecenter/${game.id}/boxscore`)
        .getContentText());

      const homeGoals = box.homeTeam?.score || 0;
      const awayGoals = box.awayTeam?.score || 0;
      const periodType = box.periodDescriptor?.periodType;
      const isOT = periodType === 'OT' || periodType === 'SO';
      const gameLabel = `${box.awayTeam?.abbrev} ${awayGoals} · ${box.homeTeam?.abbrev} ${homeGoals}`;

      if (box.homeTeam?.abbrev) teamsPlayed.add(box.homeTeam.abbrev);
      if (box.awayTeam?.abbrev) teamsPlayed.add(box.awayTeam.abbrev);

      const idToName = {};
      [box.homeTeam?.abbrev, box.awayTeam?.abbrev].forEach(abbrev => {
        if (!abbrev) return;
        try {
          const roster = JSON.parse(UrlFetchApp
            .fetch(`https://api-web.nhle.com/v1/roster/${abbrev}/20252026`)
            .getContentText());
          [...(roster.forwards||[]), ...(roster.defensemen||[]), ...(roster.goalies||[])]
            .forEach(p => { idToName[p.id] = `${p.firstName.default} ${p.lastName.default}`; });
        } catch(e) {
          agentLog('AGT-002', '⚠️', `Roster fetch failed for ${abbrev}: ${e.message}`);
        }
      });

      const pgStats = box.playerByGameStats || {};

      ['awayTeam', 'homeTeam'].forEach(side => {
        const teamAbbrev = box[side]?.abbrev;
        const teamStats  = pgStats[side] || {};
        const isHome     = side === 'homeTeam';
        const teamWon    = isHome ? homeGoals > awayGoals : awayGoals > homeGoals;

        [...(teamStats.forwards||[]), ...(teamStats.defense||[])].forEach(p => {
          if (!p.toi || p.toi === '00:00') return;
          const fullName = idToName[p.playerId] || '';
          if (!fullName) return;

          const pos = (teamStats.forwards||[]).some(f => f.playerId === p.playerId) ? 'F' : 'D';
          const stats = {
            name: fullName, team: teamAbbrev, pos,
            goals: p.goals || 0, assists: p.assists || 0,
            sog: p.sog || 0, pim: p.pim || 0,
          };

          if (stats.goals > 6 || stats.assists > 6) {
            anomalies++;
            agentLog('AGT-002', '⚠️',
              `Suspicious: ${fullName} — ${stats.goals}G ${stats.assists}A in game ${game.id}`, true);
            return;
          }

          updateSkaterStats(stats);
          checkHatTrick(stats, game.id, date);

          if (poolPlayers.has(fullName) && (stats.goals > 0 || stats.assists > 0 || stats.sog > 0)) {
            const ht = stats.goals >= 3 ? 3 : 0;
            const pimPts = pos === 'D' ? stats.pim * 0.25 : 0;
            const pts = Math.round((stats.goals + stats.assists + stats.sog * 0.11 + ht + pimPts) * 100) / 100;
            nightlyRows.push([
              fullName, teamAbbrev, pos, gameLabel,
              stats.goals, stats.assists, stats.sog, stats.pim,
              0, 0, 0, 0, 0, pts, date
            ]);
          }
        });

        (teamStats.goalies || []).forEach(g => {
          if (!g.toi || g.toi === '00:00') return;
          const fullName = idToName[g.playerId] || '';
          if (!fullName) return;

          const goalieStats = {
            name: fullName, team: teamAbbrev, pos: 'G',
            wins:     teamWon && !isOT ? 1 : 0,
            losses:   !teamWon && !isOT ? 1 : 0,
            otl:      !teamWon && isOT ? 1 : 0,
            shutouts: (g.savePctg === 1 && g.toi !== '00:00') ? 1 : 0,
            saves:    g.saves || 0,
          };
          updateGoalieStats(goalieStats);

          if (poolPlayers.has(fullName)) {
            const pts = Math.round((
              goalieStats.wins * 3 + goalieStats.losses * 1 + goalieStats.otl * 1.5 +
              goalieStats.shutouts * 2 + goalieStats.saves * 0.02
            ) * 100) / 100;
            nightlyRows.push([
              fullName, teamAbbrev, 'G', gameLabel,
              0, 0, 0, 0,
              goalieStats.wins, goalieStats.losses, goalieStats.otl,
              goalieStats.shutouts, goalieStats.saves, pts, date
            ]);
          }
        });
      });

      markGameProcessed(game.id, date,
        box.homeTeam?.abbrev, box.awayTeam?.abbrev,
        homeGoals, awayGoals);
      processed++;
    });

    if (nightlyRows.length) {
      nightlySheet.getRange(2, 1, nightlyRows.length, 15).setValues(nightlyRows);
    }

    updateIRStatuses([...teamsPlayed]);

    agentLog('AGT-002', '✅',
      `Complete — ${processed} games processed, ${skipped} skipped, ${anomalies} anomalies flagged` 
    );

    if (anomalies > 0) {
      alertCommissioner(
        'AAHL ⚠️ Stat Anomalies Detected',
        `${anomalies} suspicious stat(s) flagged on ${date}. Check AgentStatus tab.` 
      );
    }

    calculateStandings();

  } catch(err) {
    agentLog('AGT-002', '❌', `Fatal error: ${err.message}`, true);
    alertCommissioner('AAHL ❌ Stat Fetch Failed', `AGT-002 failed on ${date}: ${err.message}`);
  }
}

// ── IR Status Updater ──
function updateIRStatuses(teams) {
  const sheet = SS.getSheetByName('Players');
  if (!sheet) { agentLog('AGT-002', '⚠️', 'Players sheet not found — IR update skipped'); return; }

  const data   = sheet.getDataRange().getValues();
  const header = data[0];
  const IR_COL = 15;
  if (header[IR_COL] !== 'OnIR') {
    sheet.getRange(1, IR_COL + 1).setValue('OnIR');
  }

  const irSet = new Set();
  const teamsChecked = [];

  teams.forEach(abbrev => {
    if (!abbrev) return;
    try {
      const url      = `https://api-web.nhle.com/v1/roster/${abbrev}/current`;
      const response = UrlFetchApp.fetch(url, { muteHttpExceptions: true });
      if (response.getResponseCode() !== 200) {
        agentLog('AGT-002', '⚠️', `IR roster fetch returned ${response.getResponseCode()} for ${abbrev}`);
        return;
      }
      const roster = JSON.parse(response.getContentText());
      ['forwards', 'defensemen', 'goalies'].forEach(group => {
        (roster[group] || []).forEach(p => {
          if (p.injuryStatus && p.injuryStatus !== 'ACTIVE') {
            const firstName = (p.firstName && p.firstName.default) || '';
            const lastName  = (p.lastName  && p.lastName.default)  || '';
            const fullName  = (firstName + ' ' + lastName).trim();
            if (fullName) irSet.add(fullName);
          }
        });
      });
      teamsChecked.push(abbrev);
    } catch (e) {
      agentLog('AGT-002', '⚠️', `IR fetch exception for ${abbrev}: ${e.message}`);
    }
  });

  const irValues = [];
  for (let i = 1; i < data.length; i++) {
    const name = String(data[i][0] || '').trim();
    if (!name) { irValues.push(['']); continue; }
    irValues.push([irSet.has(name) ? 'TRUE' : 'FALSE']);
  }

  if (irValues.length > 0) {
    sheet.getRange(2, IR_COL + 1, irValues.length, 1).setValues(irValues);
  }

  agentLog('AGT-002', '✅',
    `IR statuses updated — ${irSet.size} players on IR across ${teamsChecked.length} teams checked` 
  );
}

// ── Manual trigger to seed IR column ──
function seedIRStatuses() {
  const allTeams = [
    'ANA','BOS','BUF','CAR','CBJ','CGY','CHI','COL','DAL','DET',
    'EDM','FLA','LA','MIN','MTL','NJ','NSH','NYI','NYR','OTT',
    'PHI','PIT','SEA','STL','TB','TOR','UTA','VAN','VGK','WSH','WPG','ARI'
  ];
  updateIRStatuses(allTeams);
}

function updateSkaterStats(stats) {
  const sheet = SS.getSheetByName('Players');
  const rows = sheet.getDataRange().getValues();
  const idx = rows.findIndex(r => r[0] === stats.name);
  if (idx < 0) return;
  const row = idx + 1;
  sheet.getRange(row, 6).setValue((rows[idx][5] || 0) + stats.goals);
  sheet.getRange(row, 7).setValue((rows[idx][6] || 0) + stats.assists);
  sheet.getRange(row, 8).setValue((rows[idx][7] || 0) + stats.sog);
  sheet.getRange(row, 9).setValue((rows[idx][8] || 0) + stats.pim);
}

function updateGoalieStats(stats) {
  const sheet = SS.getSheetByName('Players');
  const rows = sheet.getDataRange().getValues();
  const idx = rows.findIndex(r => r[0] === stats.name);
  if (idx < 0) return;
  const row = idx + 1;
  sheet.getRange(row, 10).setValue((rows[idx][9]  || 0) + stats.wins);
  sheet.getRange(row, 11).setValue((rows[idx][10] || 0) + stats.losses);
  sheet.getRange(row, 12).setValue((rows[idx][11] || 0) + stats.otl);
  sheet.getRange(row, 13).setValue((rows[idx][12] || 0) + stats.shutouts);
  sheet.getRange(row, 14).setValue((rows[idx][13] || 0) + stats.saves);
}

function checkHatTrick(stats, gameId, date) {
  if (stats.goals < 3) return;
  const htSheet = SS.getSheetByName('HatTricks');
  const entries = getEntries();
  const drafter = entries.find(e => e.picks.some(p => p === stats.name));
  htSheet.appendRow([
    date, stats.name, stats.team, stats.pos,
    stats.goals, stats.assists, gameId,
    drafter ? drafter.entryId : 'unowned',
    drafter ? `${drafter.firstName} ${drafter.lastName}` : '—',
    3
  ]);
  if (drafter) {
    const pts = Math.round((stats.goals + stats.assists + stats.sog * 0.11 + 3) * 100) / 100;
    sendHatTrickAlert(stats.name, stats.team, stats.goals, stats.assists, pts, drafter.teamName);
  }
}

// AGT-003 — Scoring & Standings Agent
// ══════════════════════════════════════════════
function getDivisionLeaders() {
  try {
    const res = JSON.parse(UrlFetchApp
      .fetch('https://api-web.nhle.com/v1/standings/now')
      .getContentText());
    const leaders = {};
    (res.standings || []).forEach(team => {
      const div = team.divisionName;
      if (!leaders[div]) leaders[div] = (team.teamName?.default || '').trim();
    });
    agentLog('AGT-003', '⏳', `Division leaders: ${JSON.stringify(leaders)}`);
    return leaders;
  } catch(err) {
    agentLog('AGT-003', '⚠️', `Division leaders fetch failed: ${err.message}`);
    return {};
  }
}

function calculateStandings() {
  if (!agentReady('AGT-002')) {
    agentLog('AGT-003', '⏸', 'Waiting — AGT-002 not complete. Retrying in 10 min.');
    Utilities.sleep(600000);
    if (!agentReady('AGT-002')) {
      agentLog('AGT-003', '❌', 'AGT-002 still not ready — aborting', true);
      alertCommissioner('AAHL ❌ Standings Not Updated',
        'AGT-003 aborted — AGT-002 did not complete.');
      return;
    }
  }

  agentLog('AGT-003', '⏳', 'Recalculating standings');

  const playersSheet = SS.getSheetByName('Players');
  const playerRows = playersSheet.getDataRange().getValues();
  const playerMap = {};
  playerRows.slice(1).forEach(r => {
    playerMap[r[0]] = {
      pos: r[2], goals: r[5]||0, assists: r[6]||0, sog: r[7]||0, pim: r[8]||0,
      wins: r[9]||0, losses: r[10]||0, otl: r[11]||0, shutouts: r[12]||0, saves: r[13]||0,
      hatTricks: r[14]||0
    };
  });

  const divLeaders = getDivisionLeaders();
  const entries = getEntries();

  const results = entries.map(entry => {
    let totalPts = 0, totalG = 0, totalA = 0, totalSV = 0, totalHT = 0;

    entry.picks.forEach(playerName => {
      if (!playerName) return;
      const p = playerMap[playerName];
      if (!p) return;
      let pts = 0;
      if (p.pos === 'G') {
        pts = p.wins*3 + p.losses*1 + p.otl*1.5 + p.shutouts*2 + p.saves*0.02;
        totalSV += p.saves;
      } else {
        const ht = p.hatTricks > 0 ? p.hatTricks * 3 : 0;
        const pim = p.pos === 'D' ? p.pim * 0.25 : 0;
        pts = p.goals + p.assists + p.sog * 0.11 + ht + pim;
        totalG += p.goals;
        totalA += p.assists;
        totalHT += p.hatTricks;
      }
      totalPts += pts;
    });

    let divBonus = 0;
    const divPicks = {
      'Atlantic':     entry.divAtlantic,
      'Metropolitan': entry.divMetropolitan,
      'Central':      entry.divCentral,
      'Pacific':      entry.divPacific,
    };
    Object.entries(divPicks).forEach(([div, pickedTeam]) => {
      if (!pickedTeam) return;
      const leader = divLeaders[div] || '';
      if (leader.toLowerCase() === pickedTeam.toLowerCase()) divBonus += 10;
    });
    totalPts += divBonus;

    return {
      entryId: entry.entryId, teamName: entry.teamName,
      firstName: entry.firstName, lastName: entry.lastName,
      email: entry.email,
      totalPts: Math.round(totalPts * 100) / 100,
      totalG, totalA, totalSV, totalHT,
    };
  });

  results.sort((a, b) => b.totalPts - a.totalPts || b.totalSV - a.totalSV);

  const standingsSheet = SS.getSheetByName('Standings');
  const prevRows = standingsSheet.getDataRange().getValues();
  const prevRankMap = {};
  prevRows.slice(1).forEach(r => { prevRankMap[r[1]] = parseInt(r[0]) || 0; });

  const now = new Date();
  const newRows = results.map((e, i) => {
    const rank = i + 1;
    const prev = prevRankMap[e.entryId] || rank;
    const change = prev - rank;
    return [rank, e.entryId, e.teamName, `${e.firstName} ${e.lastName}`,
      e.email, e.totalPts, prev, change, e.totalG, e.totalA, e.totalSV, e.totalHT, now];
  });

  const lastRow = standingsSheet.getLastRow();
  if (lastRow > 1) standingsSheet.getRange(2, 1, lastRow - 1, 13).clearContent();
  if (newRows.length) standingsSheet.getRange(2, 1, newRows.length, 13).setValues(newRows);

  const leader = results[0] || {};
  agentLog('AGT-003', '✅',
    `Standings updated — ${results.length} entries ranked. Leader: ${leader.teamName || '—'} with ${leader.totalPts || 0} pts` 
  );

  const cfg = CONFIG();
  const today = Utilities.formatDate(now, 'EST', 'EEEE').toUpperCase();
  if (today === (cfg['DigestDay'] || 'MONDAY')) sendWeeklyDigest(results);
}

// AGT-004 — Comms Agent
// ══════════════════════════════════════════════
function sendEntryConfirmation(entry) {
  agentLog('AGT-004', '⏳', `Sending confirmation to ${entry.email}`);

  const boxLabels = [
    'Forwards 1','Forwards 2','Forwards 3','Forwards 4','Forwards 5','Forwards 6',
    'Forwards 7','Forwards 8','Forwards 9','Forwards 10','Forwards 11','Forwards 12',
    'Defense 1','Defense 2','Defense 3','Defense 4','Defense 5','Defense 6',
    'Goalie 1','Goalie 2','Goalie 3','Goalie 4','Goalie 5','Goalie 6'
  ];

  const picks = entry.picks || [];
  const picksRows = picks.map((player, i) => {
    if (!player) return '';
    const label = boxLabels[i] || `Box ${i+1}`;
    const pos = i < 12 ? 'F' : i < 18 ? 'D' : 'G';
    const posColor = pos === 'F' ? '#2563eb' : pos === 'D' ? '#7c3aed' : '#059669';
    return `<tr style="border-bottom:1px solid #e5e7eb">
      <td style="padding:6px 10px;font-size:12px;color:#6b7280">${label}</td>
      <td style="padding:6px 10px">
        <span style="display:inline-block;background:${posColor};color:#fff;font-size:10px;font-weight:700;padding:1px 6px;border-radius:3px;margin-right:6px">${pos}</span>
        ${player}
      </td>
    </tr>`;
  }).filter(Boolean).join('');

  const divs = entry.divisions || {};

function sendWeeklyDigest(standings) {
  agentLog('AGT-004', '⏳', 'Building weekly digest');
  const allEntries = getEntries();
  const top3 = standings.slice(0, 3);
  const rows = standings.map((e, i) =>
    `<tr><td>${i+1}</td><td>${e.teamName}</td><td>${e.firstName} ${e.lastName}</td><td>${e.totalPts}</td></tr>` 
  ).join('');
  allEntries.forEach(e => {
    MailApp.sendEmail({
      to: e.email,
      subject: `🦙 AAHL Weekly Update — ${top3[0]?.teamName} leads with ${top3[0]?.totalPts} pts`,
      htmlBody: `
        <h2>This Week in AAHL</h2>
        <h3>🏆 Current Standings</h3>
        <table border="1" cellpadding="6" style="border-collapse:collapse">
          <tr><th>#</th><th>Team</th><th>Owner</th><th>Pts</th></tr>
          ${rows}
        </table>
        <p>Prize pool: $${Math.round(allEntries.length * 10 * 0.5 * 0.5)} for 1st place.</p>
      `
    });
  });
  agentLog('AGT-004', '✅', `Weekly digest sent to ${allEntries.length} participants`);
}
