How it works
Rules & Format
Pool Rules
01
$10 per entry — enter as many times as you want. E-transfer to matt.hope@rocketmail.com with your full real name in the memo.
02
All entries must be received before puck drop on opening night. Deadline: September 29, 2026 — the expected start of the 2026-27 NHL season. Exact opening game TBD when the schedule releases June 2026.
03
Select one (1) player per box — 24 boxes total covering forwards, defensemen, and goalies.
04
Points from all 24 players are combined into one total score per entry. Every goal, assist, shot, save, and penalty minute adds up over the full season.
05
No trades. Picks lock at puck drop on September 29, 2026 and cannot be changed for any reason after that.
06
Tie-breaker sequence: total points → total goals + assists → total goalie saves. If still tied after all three, the prize is split equally.
🎩
Hat Trick bonus: Any skater scoring 3 or more goals in a single game earns +3 bonus points on top of their normal goal points. A hat trick night = 3G + 2A + 8 SOG would score 3 + 3(bonus) + 2 + 0.88 = 8.88 pts. Hat tricks are also tracked in the Hat Trick Hall on the Standings page.

// AAHL APPS SCRIPT — Full Inter-Agent System

const SS = SpreadsheetApp.getActiveSpreadsheet();
const CONFIG = () => {
  const cfg = SS.getSheetByName('Config').getDataRange().getValues();
  return Object.fromEntries(cfg.map(r => [r[0], r[1]]));
};

// ── Agent Status Logger (inter-agent communication) ──
function agentLog(agentId, status, message, requiresAction = false) {
  SS.getSheetByName('AgentStatus').appendRow([
    new Date(), agentId, status, message, requiresAction ? 'Y' : 'N'
  ]);
}

// ── Check if a previous agent completed successfully ──
function agentReady(agentId) {
  const rows = SS.getSheetByName('AgentStatus').getDataRange().getValues();
  const today = Utilities.formatDate(new Date(), 'EST', 'yyyy-MM-dd');
  const todayRows = rows.filter(r =>
    Utilities.formatDate(new Date(r[0]), 'EST', 'yyyy-MM-dd') === today &&
    r[1] === agentId
  );
  if (!todayRows.length) return false;
  const last = todayRows[todayRows.length - 1];
  return last[2] === '✅';
}

// ── Alert commissioner ──
function alertCommissioner(subject, body) {
  const cfg = CONFIG();
  MailApp.sendEmail(cfg['CommissionerEmail'], subject, body);
}

// ── Duplicate game guard ──
function gameAlreadyProcessed(gameId) {
  const rows = SS.getSheetByName('ProcessedGames').getDataRange().getValues();
  return rows.some(r => String(r[0]) === String(gameId));
}

function markGameProcessed(gameId, date, home, away) {
  SS.getSheetByName('ProcessedGames').appendRow([
    gameId, date, home, away, new Date()
  ]);
}
AGT-004 — Stat Fetch AgentNightly 3 AM
function fetchNightlyStats() {
  agentLog('AGT-004', '⏳', 'Starting nightly stat fetch');
  const date = Utilities.formatDate(
    new Date(Date.now() - 86400000), 'EST', 'yyyy-MM-dd'
  ); // yesterday

  try {
    const sched = JSON.parse(UrlFetchApp
      .fetch(`https://api-web.nhle.com/v1/schedule/${date}`)
      .getContentText());

    const games = sched.gameWeek?.[0]?.games || [];
    if (!games.length) {
      agentLog('AGT-004', '✅', `No games on ${date} — nothing to process`);
      return;
    }

    let processed = 0, skipped = 0, anomalies = 0;

    games.forEach(game => {
      // ── DUPLICATE GUARD (AGT-006 dependency) ──
      if (gameAlreadyProcessed(game.id)) {
        skipped++;
        agentLog('AGT-004', '⚠️', `Game ${game.id} already processed — skipping`, false);
        return;
      }

      const box = JSON.parse(UrlFetchApp
        .fetch(`https://api-web.nhle.com/v1/gamecenter/${game.id}/boxscore`)
        .getContentText());

      // Determine game outcome for goalies
      const homeGoals = box.homeTeam?.score || 0;
      const awayGoals = box.awayTeam?.score || 0;
      const periodType = box.periodDescriptor?.periodType;
      const isOT = periodType === 'OT' || periodType === 'SO';

      // Process all skaters + goalies
      ['homeTeam','awayTeam'].forEach(side => {
        const team = box[side];
        if (!team) return;

        // Forwards & Defense
        [...(team.forwards||[]), ...(team.defense||[])].forEach(p => {
          const stats = p.toi ? {
            name:    p.name?.default || '',
            team:    team.abbrev,
            pos:     team.forwards?.includes(p) ? 'F' : 'D',
            goals:   p.goals   || 0,
            assists: p.assists  || 0,
            sog:     p.sog     || 0,
            pim:     p.pim     || 0,
          } : null;
          if (!stats) return;

          // ── SANITY CHECK (AGT-006) ──
          if (stats.goals > 6 || stats.assists > 6) {
            anomalies++;
            agentLog('AGT-006', '⚠️',
              `Suspicious stat: ${stats.name} — ${stats.goals}G ${stats.assists}A in game ${game.id}`,
              true
            );
            return; // block this player from being written
          }

          updateSkaterStats(stats);
          checkHatTrick(stats, game.id, date);
        });

        // Goalies
        (team.goalies || []).forEach(g => {
          if (!g.toi || g.toi === '00:00') return;
          const isHomeWin = homeGoals > awayGoals;
          const teamWon = (side === 'homeTeam') ? isHomeWin : !isHomeWin;
          const goalieStats = {
            name:     g.name?.default || '',
            team:     team.abbrev,
            pos:      'G',
            wins:     teamWon && !isOT ? 1 : 0,
            losses:   !teamWon && !isOT ? 1 : 0,
            otl:      !teamWon && isOT ? 1 : 0,
            shutouts: g.shotsAgainst > 0 && g.goalsAgainst === 0 ? 1 : 0,
            saves:    g.saves || 0,
          };
          updateGoalieStats(goalieStats);
        });
      });

      markGameProcessed(game.id, date,
        box.homeTeam?.abbrev, box.awayTeam?.abbrev);
      processed++;
    });

    agentLog('AGT-004', '✅',
      `Complete — ${processed} games processed, ${skipped} skipped, ${anomalies} anomalies flagged`
    );

    if (anomalies > 0) {
      alertCommissioner(
        'AAHL ⚠️ Stat Anomalies Detected',
        `${anomalies} suspicious stat(s) were flagged and blocked on ${date}. Check the AgentStatus tab.`
      );
    }

    // ── HAND OFF TO AGT-005 ──
    calculateStandings();

  } catch(err) {
    agentLog('AGT-004', '❌', `Fatal error: ${err.message}`, true);
    alertCommissioner('AAHL ❌ Stat Fetch Failed', `AGT-004 failed on ${date}: ${err.message}`);
  }
}
AGT-005 — Scoring & Standings AgentRuns after AGT-004
function calculateStandings() {
  // ── CHECK AGT-004 COMPLETED FIRST ──
  if (!agentReady('AGT-004')) {
    agentLog('AGT-005', '⏸', 'Waiting — AGT-004 not complete. Will retry in 10 min.');
    Utilities.sleep(600000);
    if (!agentReady('AGT-004')) {
      agentLog('AGT-005', '❌', 'AGT-004 still not ready after retry — aborting', true);
      alertCommissioner('AAHL ❌ Standings Not Updated',
        'AGT-005 could not run because AGT-004 did not complete successfully.');
      return;
    }
  }

  agentLog('AGT-005', '⏳', 'Recalculating standings');

  try {
    const players = getPlayerMap();
    const entries = getEntries();

    const ranked = entries.map(e => {
      const pts = Math.round(
        e.picks.reduce((s, name) => s + (players[name]?.pts || 0), 0) * 100) / 100;
      const goals   = e.picks.reduce((s, n) => s + (players[n]?.goals   || 0), 0);
      const assists = e.picks.reduce((s, n) => s + (players[n]?.assists  || 0), 0);
      const saves   = e.picks.reduce((s, n) => s + (players[n]?.saves    || 0), 0);
      const hts     = e.picks.reduce((s, n) => s + (players[n]?.hatTricks|| 0), 0);
      return { ...e, pts, goals, assists, saves, hts };
    }).sort((a, b) =>
      b.pts - a.pts ||
      (b.goals + b.assists) - (a.goals + a.assists) ||  // tiebreaker 1
      b.saves - a.saves                                  // tiebreaker 2
    );

    writeStandings(ranked);

    agentLog('AGT-005', '✅',
      `Standings updated — ${ranked.length} entries ranked. Leader: ${ranked[0]?.teamName} with ${ranked[0]?.pts} pts`
    );

    // ── HAND OFF TO AGT-007 (weekly digest check) ──
    const day = new Date().getDay(); // 1 = Monday
    if (day === 1) sendWeeklyDigest(ranked);

  } catch(err) {
    agentLog('AGT-005', '❌', `Standings calculation failed: ${err.message}`, true);
    alertCommissioner('AAHL ❌ Standings Failed', err.message);
  }
}

function calcPoints(p) {
  if (p.pos === 'G') {
    return Math.round((
      (p.wins * 3) + (p.losses * 1) + (p.otl * 1.5) +
      (p.shutouts * 2) + (p.saves * 0.02)
    ) * 100) / 100;
  }
  const htBonus = p.hatTricks > 0 ? p.hatTricks * 3 : 0;
  const pimPts  = p.pos === 'D' ? p.pim * 0.25 : 0;
  return Math.round((
    (p.goals * 1) + (p.assists * 1) + (p.sog * 0.11) + htBonus + pimPts
  ) * 100) / 100;
}

// ── Weekly digest (fires every Monday via AGT-005 handoff) ──
function sendWeeklyDigest(standings) {
  // ── CHECK AGT-005 COMPLETED ──
  if (!agentReady('AGT-005')) {
    agentLog('AGT-007', '⏸', 'Holding digest — AGT-005 standings not ready', false);
    return;
  }

  agentLog('AGT-007', '⏳', 'Building weekly digest');
  const cfg = CONFIG();
  const allEntries = getEntries();

  const top3 = standings.slice(0, 3);
  const rows = standings.map((e, i) =>
    `${i+1}${e.teamName}${e.firstName} ${e.lastName}${e.pts}`
  ).join('');

  allEntries.forEach(e => {
    MailApp.sendEmail({
      to: e.email,
      subject: `🦙 AAHL Weekly Update — ${top3[0].teamName} leads with ${top3[0].pts} pts`,
      htmlBody: `
        
This Week in AAHL

        
🏆 Current Standings

        
          ${rows}
        

          
#TeamOwnerPts

        
Prize pool: $${Math.round(allEntries.length * 10 * 0.5 * 0.5)} for 1st place.

        
Check live standings anytime at your pool link.

      `
    });
  });

  agentLog('AGT-007', '✅', `Weekly digest sent to ${allEntries.length} participants`);
}

// ── Receives web form POST and writes to Entries sheet ──
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const cfg = CONFIG();

    if (cfg['PicksLocked'] === true || cfg['PicksLocked'] === 'TRUE') {
      return ContentService.createTextOutput(
        JSON.stringify({ success: false, error: 'Picks are locked.' })
      ).setMimeType(ContentService.MimeType.JSON);
    }

    const sheet = SS.getSheetByName('Entries');
    const id = 'AAHL-' + Date.now();

    sheet.appendRow([
      id,
      data.firstName, data.lastName,
      data.email, data.teamName,
      'N', // payment not yet confirmed
      new Date(),
      ...data.picks // 24 player names
    ]);

    agentLog('AGT-002', '✅', `New entry: ${data.teamName} by ${data.firstName} ${data.lastName}`);

    // ── HAND OFF TO AGT-007 ──
    sendEntryConfirmation(data);

    return ContentService.createTextOutput(
      JSON.stringify({ success: true, entryId: id })
    ).setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    agentLog('AGT-002', '❌', `Entry submission failed: ${err.message}`, true);
    return ContentService.createTextOutput(
      JSON.stringify({ success: false, error: err.message })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}
