Apps Script — nightly scoring

function fetchNightlyStats() {
  const date = Utilities.formatDate(new Date(), 'EST', 'yyyy-MM-dd');
  const sched = JSON.parse(UrlFetchApp
    .fetch(`https://api-web.nhle.com/v1/schedule/${date}`)
    .getContentText());

  (sched.gameWeek?.[0]?.games || []).forEach(g => {
    const box = JSON.parse(UrlFetchApp
      .fetch(`https://api-web.nhle.com/v1/gamecenter/${g.id}/boxscore`)
      .getContentText());
    updatePlayerStats(box);
  });
}

function calcPoints(p) {
  return p.position === 'G'
    ? (p.wins * 2) + (p.shutouts * 2)
    : p.goals + p.assists;
}

function calculateStandings() {
  const players = getPlayerMap();
  const entries = getEntries();

  return entries
    .map(e => ({
      ...e,
      total: e.picks.reduce((s, name) =>
        s + (players[name]?.pts || 0), 0)
    }))
    .sort((a, b) => b.total - a.total);
}
